// Job statuses
const JOB_STATUS = {
  QUEUED: "queued",
  PROCESSING: "processing",
  COMPLETED: "completed",
  FAILED: "failed",
  TIMEOUT: "timeout"
};

// ✅ Non-blocking execution function (worker)
const executeSubmission = async (jobId, code, language_id, stdin = "") => {
  try {
    jobStore.set(jobId, {
      status: JOB_STATUS.PROCESSING,
      token: null,
      result: null,
      error: null,
      createdAt: Date.now()
    });

    console.log(`[WORKER] Job ${jobId} - Submitting to Judge0`);

    // Step 1: Submit to Judge0
    const submit = await axios.post(
      `${JUDGE0_URL}/submissions`,
      { source_code: code, language_id, stdin },
      { params: { base64_encoded: false, fields: "token" } }
    );

    const token = submit.data.token;
    if (!token) throw new Error("No token from Judge0");

    console.log(`[WORKER] Job ${jobId} - Token: ${token}`);

    // Update job with token
    const job = jobStore.get(jobId);
    job.token = token;

    // Step 2: Poll until complete
    const result = await pollSubmission(token, { maxRetries: 60, intervalMs: 500 });

    if (!result.done) {
      jobStore.set(jobId, { ...job, status: JOB_STATUS.TIMEOUT, error: "Execution timeout" });
      console.error(`[WORKER] Job ${jobId} - Timeout`);
      return;
    }

    // Step 3: Store result
    const payload = buildPayload(result.data);
    jobStore.set(jobId, {
      status: JOB_STATUS.COMPLETED,
      token,
      result: payload,
      error: null,
      createdAt: Date.now()
    });

    console.log(`[WORKER] Job ${jobId} - ✓ Completed`);
  } catch (err) {
    console.error(`[WORKER] Job ${jobId} - ✗ Error:`, err.message);
    jobStore.set(jobId, {
      status: JOB_STATUS.FAILED,
      token: null,
      result: null,
      error: err.message,
      createdAt: Date.now()
    });
  }
};

// Cleanup old jobs (every 5 minutes)
setInterval(() => {
  const now = Date.now();
  const maxAge = 30 * 60 * 1000; // 30 minutes
  for (const [jobId, job] of jobStore.entries()) {
    if (now - job.createdAt > maxAge) {
      jobStore.delete(jobId);
      console.log(`[CLEANUP] Deleted job ${jobId}`);
    }
  }
}, 5 * 60 * 1000);

const buildPayload = (data) => {
  const combinedOutput = data.stdout || data.stderr || data.compile_output || "";
  
  // Extract status carefully - it may be an object or string
  let statusDesc = data.status?.description || data.status || "Unknown";
  if (typeof statusDesc === "object") {
    statusDesc = statusDesc.description || "Unknown";
  }
  
  let statusId = data.status?.id;
  if (typeof data.status === "object" && !statusId) {
    statusId = data.status.id;
  }
  
  return {
    token: data.token,
    status: statusDesc,
    status_id: statusId,
    stdout: data.stdout || "",
    stderr: data.stderr || "",
    compile_output: data.compile_output || "",
    message: data.message || "",
    time: data.time,
    memory: data.memory,
    output: combinedOutput // backward-friendly field for simple UIs
  };
};

// Small sleep helper for polling intervals.
const delay = (ms) => new Promise((resolve) => setTimeout(resolve, ms));

// Poll Judge0 until the submission finishes or we hit retry limits.
const pollSubmission = async (token, { maxRetries = 30, intervalMs = 800 } = {}) => {
  let attempt = 0;
  console.log(`[POLL] Starting poll for token: ${token}`);
  
  while (attempt < maxRetries) {
    try {
      const resp = await axios.get(`${JUDGE0_URL}/submissions/${token}`, {
        params: {
          base64_encoded: false,
          fields: "token,status,status_id,stdout,stderr,compile_output,message,time,memory"
        }
      });

      // Judge0 returns status as { id: number, description: string }
      const statusId = resp.data.status?.id;
      const statusDesc = resp.data.status?.description;
      
      console.log(`[POLL] Attempt ${attempt + 1}/${maxRetries}: status=${statusDesc} (id=${statusId})`);

      // Status 1 = In Queue, 2 = Processing
      // Any other status means done
      if (statusId !== 1 && statusId !== 2) {
        console.log(`[POLL] Execution complete. Status: ${statusDesc}`);
        return { done: true, data: resp.data };
      }

      attempt += 1;
      await delay(intervalMs);
    } catch (err) {
      console.error(`[POLL] Error during polling attempt ${attempt + 1}:`, err.message);
      attempt += 1;
      await delay(intervalMs);
    }
  }

  // Timed out
  console.error(`[POLL] Timeout after ${maxRetries} attempts`);
  return { done: false, data: null };
};

app.get("/job/:jobId", (req, res) => {
  const { jobId } = req.params;

  const job = jobStore.get(jobId);
  if (!job) {
    return res.status(404).json({ error: "Job not found" });
  }

  // If still processing, return status only
  if (job.status === JOB_STATUS.QUEUED || job.status === JOB_STATUS.PROCESSING) {
    return res.json({
      job_id: jobId,
      status: job.status,
      message: job.status === JOB_STATUS.QUEUED ? "Queued..." : "Executing..."
    });
  }

  // If completed, return full result
  if (job.status === JOB_STATUS.COMPLETED) {
    return res.json({
      job_id: jobId,
      status: job.status,
      result: job.result
    });
  }

  // If failed or timeout
  res.status(400).json({
    job_id: jobId,
    status: job.status,
    error: job.error
  });
});

function App() {
  const [code, setCode] = useState(TEMPLATES[54]); // Start with C++
  const [languageId, setLanguageId] = useState(54);
  const [stdin, setStdin] = useState("");
  const [output, setOutput] = useState(null);
  const [isRunning, setIsRunning] = useState(false);
  const [error, setError] = useState("");
  const [jobId, setJobId] = useState(null);
  const [pollInterval, setPollInterval] = useState(null);

  // Handle language change and load template
  const handleLanguageChange = (newLangId) => {
    const langId = Number(newLangId);
    setLanguageId(langId);
    setCode(TEMPLATES[langId] || "");
    setOutput(null);
    setError("");
  };

  // Poll job status
  const pollJob = async (jId) => {
    try {
      const res = await fetch(`${API_BASE}/job/${jId}`);
      const data = await res.json();

      if (data.status === "queued" || data.status === "processing") {
        // Still running, keep polling
        return;
      }

      // Job finished
      if (data.status === "completed") {
        setOutput(data.result);
        setError("");
      } else {
        setError(data.error || "Job failed");
      }

      // Stop polling
      clearInterval(pollInterval);
      setIsRunning(false);
      setJobId(null);
    } catch (err) {
      console.error("Polling error:", err);
    }
  };

  // Submit code
  const runCode = async () => {
    setIsRunning(true);
    setError("");
    setOutput(null);
    setJobId(null);

    try {
      // Submit and get job ID (non-blocking)
      const res = await fetch(`${API_BASE}/run`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ code, language_id: languageId, stdin })
      });

      const data = await res.json();

      if (!res.ok) {
        setError(data.error || "Submission failed");
        setIsRunning(false);
        return;
      }

      const jId = data.job_id;
      setJobId(jId);
      console.log("Job submitted:", jId);

      // Start polling for results
      const interval = setInterval(() => pollJob(jId), 500);
      setPollInterval(interval);
    } catch (err) {
      setError(err.message || "Request failed");
      setIsRunning(false);
    }
  };

  return (
    <div className="page">
      <header className="hero">
        <div>
          <p className="eyebrow">MVP · Judge0 Wrapper</p>
          <h1>Run code in the sandbox.</h1>
          <p className="lede">
            Paste code, pick a language, and see stdout, stderr, or compile errors straight from Judge0.
          </p>
        </div>
        <button className="run" onClick={runCode} disabled={isRunning}>
          {isRunning ? `Running... (Job: ${jobId?.slice(0, 8)})` : "Run"}
        </button>
      </header>

      <section className="grid">
        <div className="card">
          <div className="field">
            <label htmlFor="language">Language</label>
            <select
              id="language"
              value={languageId}
              onChange={(e) => handleLanguageChange(e.target.value)}
            >
              {LANGUAGES.map((lang) => (
                <option key={lang.id} value={lang.id}>
                  {lang.name}
                </option>
              ))}

  const connections = sseConnections.get(userId);

  console.log(`[SSE] Broadcasting to user ${userId} (${connections.length} connections)`);

      res.write(message);

  });

          if (jobData.userId) {
            console.log(`[SSE] Sending job-complete event for job ${jobId}`);
            broadcastToUser(jobData.userId, {
              jobId,
              result: payload
            });
          }

          clearInterval(pollInterval);
          tokenToJobId.delete(token);
        } else if (attempts >= maxAttempts) {
          // Timeout
          console.log(`[POLL] Job ${jobId} - ✗ TIMEOUT (30s)`);
          const jobData = jobStore.get(jobId);
          jobData.status = JOB_STATUS.TIMEOUT;
          jobData.error = "Execution timeout (30 seconds)";

          if (jobData.userId) {
            broadcastToUser(jobData.userId, {
              jobId,
              error: "Timeout"
            });
          }

          clearInterval(pollInterval);
          tokenToJobId.delete(token);
        }
      } catch (pollErr) {
        console.error(`[POLL] Job ${jobId} - Polling error:`, pollErr.message);
        if (attempts >= maxAttempts) {
          clearInterval(pollInterval);
        }
      }
    }, 500);

  } catch (err) {
    console.error(`[WORKER] Job ${jobId} - ✗ Error:`, err.message);
    const jobData = jobStore.get(jobId);
    jobData.status = JOB_STATUS.FAILED;
    jobData.error = err.message;
  }
};

// Cleanup old jobs (every 5 minutes)
setInterval(() => {
  const now = Date.now();
  const maxAge = 30 * 60 * 1000; // 30 minutes
  for (const [jobId, job] of jobStore.entries()) {
    if (now - job.createdAt > maxAge) {
      jobStore.delete(jobId);
      console.log(`[CLEANUP] Deleted job ${jobId}`);
    }
  }
}, 5 * 60 * 1000);

const buildPayload = (data) => {
  const combinedOutput = data.stdout || data.stderr || data.compile_output || "";
  
  // Extract status carefully - it may be an object or string
  let statusDesc = data.status?.description || data.status || "Unknown";
  if (typeof statusDesc === "object") {
    statusDesc = statusDesc.description || "Unknown";
  }
  
  let statusId = data.status?.id;
  if (typeof data.status === "object" && !statusId) {
    statusId = data.status.id;
  }
  
  return {
    token: data.token,
    status: statusDesc,
    status_id: statusId,
    stdout: data.stdout || "",
    stderr: data.stderr || "",
    compile_output: data.compile_output || "",
    message: data.message || "",
    time: data.time,
    memory: data.memory,
    output: combinedOutput // backward-friendly field for simple UIs
  };
};

// 📡 SSE Endpoint - Frontend connects here for real-time updates
app.get("/events", (req, res) => {
  const userId = req.query.userId || "anonymous";
  
  console.log(`[SSE] New connection from user: ${userId}`);
  
  // Set SSE headers
  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");
  res.setHeader("Connection", "keep-alive");
  res.setHeader("Access-Control-Allow-Origin", "*");
  
  // Add connection to user's list
  if (!sseConnections.has(userId)) {
    sseConnections.set(userId, []);
  }
  sseConnections.get(userId).push(res);
  
  // Send initial connection confirmation
  res.write(`event: connected\ndata: {"userId":"${userId}"}\n\n`);
  
  // Clean up on disconnect
  req.on("close", () => {
    console.log(`[SSE] Connection closed for user: ${userId}`);
    const connections = sseConnections.get(userId);
    if (connections) {
      const filtered = connections.filter(r => r !== res);
      if (filtered.length === 0) {
        sseConnections.delete(userId);
      } else {
        sseConnections.set(userId, filtered);
      }
    }
  });
});

  // Return job ID immediately
  res.json({
    job_id: jobId,
    status: JOB_STATUS.QUEUED,
    message: "Job queued. Poll /job/{job_id} for results."
  });
});


// 2️⃣ NEW: Fetch job status and result
app.get("/job/:jobId", (req, res) => {
  const { jobId } = req.params;

  console.log(`[POLL] /job/${jobId} - Status check`);
  
  const job = jobStore.get(jobId);
  if (!job) {
    console.log(`[POLL] Job ${jobId} - ✗ NOT FOUND`);
    return res.status(404).json({ error: "Job not found" });
  }

  console.log(`[POLL] Job ${jobId} - Status: ${job.status}`);

  // If still processing, return status only
  if (job.status === JOB_STATUS.QUEUED || job.status === JOB_STATUS.PROCESSING) {
    console.log(`[POLL] Job ${jobId} - Still running...`);
    return res.json({
      job_id: jobId,
      status: job.status,
      message: job.status === JOB_STATUS.QUEUED ? "Queued..." : "Executing..."
    });
  }

  // If completed, return full result
  if (job.status === JOB_STATUS.COMPLETED) {
    console.log(`[POLL] Job ${jobId} - ✓ COMPLETED - Returning result`);
    return res.json({
      job_id: jobId,
      status: job.status,
      result: job.result
    });
  }

  // If failed or timeout
  console.log(`[POLL] Job ${jobId} - ✗ ${job.status.toUpperCase()}`);
  res.status(400).json({
    job_id: jobId,
    status: job.status,
    error: job.error
  });
});
